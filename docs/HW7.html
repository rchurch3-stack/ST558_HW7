Mutating Data Table

  • Convert the Date column into an actual date.

  • Turn the character variables (Seasons, Holiday, and Functioning Day) into factors.

  • Rename the all the variables to have easy to use names

fixed_bike_data <- bike_data |>
  mutate(
    season = as.factor(Seasons),
    holiday = as.factor(Holiday),
    functioning = as.factor(`Functioning Day`),
    Date = dmy(Date)
  ) |>
  rename(
    "date" = "Date",
    "rented_bike_count" = "Rented Bike Count",
    "hour" = "Hour",
    "temp" = "Temperature(°C)",
    "humid" = "Humidity(%)",
    "wind" = "Wind speed (m/s)",
    "vis" = "Visibility (10m)",
    "dwpt" = "Dew point temperature(°C)",
    "solar_rad" = "Solar Radiation (MJ/m2)",
    "rain" = "Rainfall(mm)",
    "snow" = "Snowfall (cm)",
  )

Recipe 1

  • We will first create a weekend or weekday variable called day_type

  • Set Dummy variables for day_type, season, and holiday

  • Normalize all numeric predictors. Use all of them linerly.

recipe1 <- recipe(rented_bike_count ~ ., data = daily_bike_data) |>
  step_date(date, features = c("dow")) |>
  step_mutate(

      date_dow %in% c("Sat", "Sun"),
      "Weekend",
      "Weekday"
    ))
  ) |>
  step_rm(date, date_dow) |>
  step_dummy(season, holiday, day_type) |>
  step_normalize(all_numeric_predictors(), -all_outcomes())

recipe3 <- recipe(rented_bike_count ~ ., data = daily_bike_data) |>
  step_date(date, features = "dow") |>
