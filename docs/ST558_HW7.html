Next, read the CSV file into R using the read_csv() function from the readr package.

bike <- read_csv(
  "Data/SeoulBikeData.csv",
  locale = locale(encoding = "Windows-1252"),
  show_col_types = FALSE
) |>
  clean_names()

# convert few of variables to factors
bike <- bike |>
  mutate(
    date = as.Date(date, format = "%d/%m/%Y"),
    across(
      c(seasons, holiday, functioning_day),
      as.factor
    )
  )
